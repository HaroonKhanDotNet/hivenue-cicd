function _buildFilters(req) {
    const filters = {};
 
    if (req.query.provider === 'facebook' && req.query.app_user_id) {
        filters.app_user_id = req.query.app_user_id;
    } else if (req.query.provider === 'GOOGLE' && req.query.app_user_id) {
        filters.app_user_id = req.query.app_user_id;
    }
 
    if (req.query.email) {
        filters.email = req.query.email.toLowerCase();
    }
 
    if (req.params.id) {
        filters._id = new ObjectId(req.params.id);
    }
    filters.isActive = true;
 
    if (typeof req.query.userCode !== 'undefined') {
        if (req.query.userCode === '') {
            return errors.badRequest('Please enter a usercode');
        }
 
        filters.userCode = req.query.userCode;
    }
 
    return { filters };
}

async function _updateUserReferences(user) {
    const updateOwner = {
        filter: { 'owner.userId': user._id },
        values: {
            'owner.name': user.first_name + ' ' + user.last_name,
            'owner.photo': user.customImage ? user.customImage : user.image,
        },
    };
 
    await Apartment.updateMany(updateOwner.filter, updateOwner.values, {

        {
            $set: {
                'rooms.$.userName': user.first_name + ' ' + user.last_name,
                'rooms.$.userPhoto': user.customImage ? user.customImage : user.image,
            },
        },

}
 
exports.getUser = async (req, res) => {
    // If the user has specified the target user id in the query,
    // then only allow that user id to be himself - don't allow
    // the user to get other users' data
    // The exception is, if the user is an admin, then return the specified user
    if (!isAdminEmail(req.user.email) && req.params.id) {
        req.params.id = req.authUserId;
    }
 
    let user = await User.findOne({ _id: req.params.id })
        .select(sensitiveFields)
        .catch(() => {
            return errors.internalServerError('Failed to get user');
        });
 
    if (!user) return errors.notFound('User does not exist');
 
    if (!user.isActive) return errors.forbidden('User has been blocked');
    const documentFields = ['identity', 'residency', 'poi', 'otherDocument', 'pos', 'poe'];
    documentFields.forEach((field) => {
        if (field in user && user[field].length > 1) {
            user[field] = [user[field][user[field].length - 1]];
        }
    });
 
    // Matching Data
    let matchingData = await MatchingData.findOne({
        type: 'genuine',
        user_id: new ObjectId(req.params.id),
    });
    if (matchingData) {
        user.aboutme = matchingData.aboutme;
        user.keywords = matchingData.categorized_keywords; // why this is empty
    } else {
        user.aboutme = '';
        user.keywords = [];
    }
 
    return res.json(user);
};

exports.verifyToken = async (req, res, next) => {
    const data = req.query.provider !== undefined ? req.query : req.body;
 
    if (!data.provider || !data.token) {
        return errors.unauthorized('You must provide provider and token for authentication');
    }
 
    if (!['facebook', 'GOOGLE'].includes(data.provider)) {
        return errors.unauthorized('Unknown provider');
    }
 
    if (data.provider === 'facebook') {
        // Verify the provided token
        let verifyRes;
        try {
            verifyRes = await httpGet('https://graph.facebook.com/me?fields=email&access_token=' + data.token);
        } catch (e) {
            return errors.unauthorized('Unable to verify the token');
        }
 
        // Validate the provided token
        if (
            !verifyRes.data.email ||
            !verifyRes.data.id ||
            verifyRes.data.email.toLowerCase() !== data.email.toLowerCase() ||
            verifyRes.data.id !== data.app_user_id
        ) {
            return errors.unauthorized('Unable to verify the token');
        }
        req.body.email = verifyRes.data.email;
    } else if (data.provider === 'GOOGLE') {
        // Obtain user info from the decoded token
        const payload = await exports.verifyGoogleToken(data.token);
        req.body.first_name = payload.given_name;
        req.body.last_name = payload.family_name;
        req.body.email = payload.email;
        req.body.image = payload.picture;
        req.body.app_user_id = payload.sub;
        data.app_user_id = payload.sub;
        req.query.email = payload.email;
    }
 
    log.info('User ' + data.app_user_id + ' has login with ' + data.provider);
 
    next();
};

exports.verifyEmailPassword = async (req, res) => {
    const data = req.body;
 
    if (data.provider !== 'email' || !data.email || !data.password) {
        return errors.unauthorized('No credentials provided');
    }
 
    data.email = data.email.toLowerCase();
    const user = await User.findOne({ email: data.email });
 
    if (!user) {
        return errors.unauthorized('Unable to verify email/password combination');
    }
 
    if (user.emailConfirmPending && user.emailConfirmPending === true) {
        return errors.unauthorized('Please confirm the email address of your account');
    }
 
    // Validate the password hash
    // Nnote that, data.token is user provided password
    if (!user.passwordHash || !user.passwordSalt) {
        return errors.unauthorized('Unable to verify email/password combination');
    }
 
    // Check failed attempts
    // 11 attempts -> block 30 seconds
    // After that, allow the user to try 3 times in a row every time. Then,
    // every (3n) more attempt -> block (30 + (n / 3) * 30) seconds
    if (user.passwordAttempts > 11) {
        if ((user.passwordAttempts - 11) % 3 === 0) {
            if (Date.now() - user.passwordLastAttempt < (30 + (user.passwordAttempts - 11) * 10) * 1000) {
                return errors.forbidden('Too many failed attempts, please try again later');
            }
        }
    } else if (user.passwordAttempts === 10) {
        // Block 30 seconds for the 11th try
        if (Date.now() - user.passwordLastAttempt < 30 * 1000) {
            return errors.forbidden('Too many failed attempts, please try again later');
        }
    }
 
    const hash = crypto.pbkdf2Sync(data.password, user.passwordSalt, 1000, 64, 'sha512').toString('hex');
    if (hash === user.passwordHash) {
        // Clear failed attempts
        user.updateFields({ passwordAttempts: 0 });
        user.updateFields({ passwordLastAttempt: new Date(1970) });
        const token = await AuthController.generateLoginToken(user._id);
 
        log.info('User ' + user.id + ' has login with email/password');
 
        return res.json({
            user: await _saveUser(user),
            token: token,
        });
    } else {
        // Increase the number of attempts, and save the attempt time
        user.updateFields({ passwordAttempts: user.passwordAttempts + 1 });
        user.updateFields({ passwordLastAttempt: new Date() });
        await user.save();
 
        return errors.unauthorized('Unable to verify email/password combination');
    }
};

exports.getAllUsers = async (req, res) => {
    let { filters } = _buildFilters(req);
 
    try {
        // First match user by ID (either Facebook or Google user ID) ONLY
        delete filters.email;
        let users = await User.find(filters).select(sensitiveFields);
 
        if (req.query.email !== undefined && typeof req.query.email == 'string') {
            if (users.length == 0) {
                // If no user is found by ID, then attempt to match user by email
                if (req.query.email) {
                    let newFilter = {};
                    req.query.email = req.query.email.toLowerCase();
                    newFilter['email'] = req.query.email;
 
                    let userByEmail = await User.findOne(newFilter).select(sensitiveFields);
                    if (!userByEmail) {
                        return res.json([]);
                    }
                    const token = await AuthController.generateLoginToken(userByEmail._id);
                    return res.json({
                        user: userByEmail,
                        token: token,
                    });
                }
            } else if (users.length == 1) {
                // If a unique user is found by ID, but that user doesn't have an email, update its email
                if (req.query.email) {
                    if (users[0].email === undefined) {
                        req.query.email = req.query.email.toLowerCase();
                        users[0].updateFields({
                            email: req.query.email,
                        });
                        await _saveUser(users[0]);
                        users[0].email = req.query.email;
                    }
                }
            } else {
                // If multiple users are found by ID, then return an error
                return errors.badRequest('Cannot identify the user');
            }
        } else {
            return errors.badRequest('No email provided');
        }
 
        const token = await AuthController.generateLoginToken(users[0]._id);
        return res.json({
            user: users[0],
            token: token,
        });
    } catch (err) {
        return errors.badRequest(err);
    }
};

exports.sendVerificationCode = async (req, res) => {
    const user = await User.findOne({ email: req.body.email });
    if (!user) {
        return res.status(400).json({ message: 'cannot find user' });
    }
    // make sure that the email is granted with admin access
    const admin = await Admin.findOne({ userId: user._id });
    if (!admin) {
        return res.status(400).json({ message: "You're not granted with admin access" });
    }
    // make sure that one user cannot request verification code more than twice in 1 minute
    const token = await Token.findOne({
        user_id: user._id,
        type: 'email',
        create_time: {
            $gte: Date.now() - 60 * 1000,
        },
    });
    if (!token) {
        // generate verification code and save it in db
        const verificationCode = await AuthController.generateEmailVerificationCode(user._id);
        sendVerificationCodeEmail(req.body.email, verificationCode.value);
        return res.status(200).json({});
    } else {
        return res.status(400).json({ message: 'You can only request verification code for ONE time in ONE minute' });
    }
};
 
exports.verifyDocumentOwnership = async (req, res, next) => {
    const { filename, type } = req.params;
    const user = req.user;
 
    const documentTypes = {
        identities: 'identity',
        'proofs-of-income': 'poi',
        'proof-of-study': 'pos',
        'proof-of-employment': 'poe',
        others: 'otherDocument',
        residency: 'residency',
    };
 
    const userDocuments = user[documentTypes[type]];
    const isUserOwnedFile = userDocuments.some((doc) => doc.filename === filename);
 
    if (!isUserOwnedFile) {
        return errors.badRequest('Invalid file requested');
    }
 
    next();
};
