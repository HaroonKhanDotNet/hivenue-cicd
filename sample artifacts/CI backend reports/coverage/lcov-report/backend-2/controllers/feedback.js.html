const feedbackModels = require('../models/feedback');
const UserModel = require('../models/user');
const errors = require('../utils/error');
const { USER_STATUS, PENDING_FEEDBACK_STATUS } = require('../constants/constants');
const { ObjectId } = require('mongoose/lib/types');
 
/**
 * Get the active feedback form with the questions filtered by user status
 * @param {params: {userStatus}} req
 * @param {*} res
 * @returns the active feedback form
 */
const getActiveFeedbackForm = async (req, res) => {
    if (!req.params?.userStatus) return errors.badRequest('id required');
 
    const userStatus = req.params.userStatus.toLowerCase();
 
    if (!Object.values(USER_STATUS).includes(userStatus)) return errors.badRequest('Invalid user status');
 
    const form = await feedbackModels.formModel.findOne({}, null, {
        sort: {
            active: -1,
        },
    });
 
    if (!form) return errors.notFound('There is no questions form');
 
    form.questions = form.questions.filter((q) => q.userStatus === userStatus || q.userStatus === USER_STATUS.BOTH);
    res.status(200).send(form);
};
 
/**
 * Get the pending feedback for a specific user
 * @param {} req
 * @param {} res
 * @returns A list of the pending feedback
 */
const getPendingFeebacks = async (req, res) => {
    //Find the user
    const user = await UserModel.findOne({ _id: req.userId });
    if (!user) return errors.unauthorized('You are not logged in');
 
    const pendingFeedbacks = await feedbackModels.pendingFeedbackModel
        .find({ status: PENDING_FEEDBACK_STATUS.PENDING, userId: new ObjectId(req.userId) })
        .populate({
            path: 'rentRequest',
            model: 'pendingRentRequest',
            populate: { path: 'apartmentId', model: 'apartment' },
        });
 
    const response = pendingFeedbacks.map((pf) => {
        return {
            aptName: pf.rentRequest.apartmentId.name,
            pendingFeedbackId: pf._id,
        };
    });
 
    return res.status(200).send(response);
};
 
/**
 * Get a pending feedback by id for checkup in feedback page
 * @param {params: {id: pendingfeebackId}} req
 * @param {} res
 * @returns the user ID and status, the rent request ID and the apartment ID
 */
const findPendingFeedbackById = async (req, res) => {
    if (!req.params.id) return errors.badRequest('Pending feedback ID required');
    if (!ObjectId.isValid(req.params.id)) return errors.badRequest('Invalid pending feedback ID');
 
    //Find the user
    const user = await UserModel.findOne({ _id: req.userId });
    if (!user) return errors.unauthorized('You are not logged in');
 
    // Find the pending request
    const pendingFeedback = await feedbackModels.pendingFeedbackModel
        .findOne({
            _id: new ObjectId(req.params.id),
        })
        .populate({
            path: 'rentRequest',
            model: 'pendingRentRequest',
            populate: { path: 'apartmentId', model: 'apartment' },
        });
 
    if (!pendingFeedback) return errors.notFound('Pending feedback not found.');
 
    // check if the user has the right to access this pending
    if (pendingFeedback.userId.toString() != req.userId) return errors.badRequest('wrongUser');
 
    // Check if the pending feedback still valid
    if (pendingFeedback.status === PENDING_FEEDBACK_STATUS.ANSWERED) return errors.badRequest('alreadyAnswered');
 
    // Check if apartment exists
    if (!pendingFeedback.rentRequest?.apartmentId) return errors.badRequest('Apartment not found');
 
    res.status(200).send({
        userId: req.userId,
        userStatus: pendingFeedback.userStatus,
        rentRequestId: pendingFeedback.rentRequest._id.toString(),
        apartmentId: pendingFeedback.rentRequest.apartmentId._id.toString(),
    });
};
 
/**
 * Submit the user's feedback answers
 * @param {params: {answers, userId, apartmentId, formId, userStatus}} req
 * @param {} res
 * @returns the saved answers
 */
const submitFormAnswer = async (req, res) => {
    if (!req.body?.answers) return errors.badRequest('answers required');
    if (!req.body?.userId) return errors.badRequest('user ID required');
    if (!ObjectId.isValid(req.body.userId)) return errors.badRequest('Invalid user ID');
    if (!req.body?.apartmentId) return errors.badRequest('appartment ID required');
    if (!ObjectId.isValid(req.body.apartmentId)) return errors.badRequest('Invalid appartment ID');
    if (!req.body?.formId) return errors.badRequest('form ID required');
    if (!ObjectId.isValid(req.body.formId)) return errors.badRequest('Invalid form ID');
    if (!req.body?.rentRequestId) return errors.badRequest('rent request ID required');
    if (!ObjectId.isValid(req.body.rentRequestId)) return errors.badRequest('Invalid rent request ID');
    if (!req.body?.userStatus) return errors.badRequest('user status required');
 
    const newAnswers = feedbackModels.answersModel({
        userId: req.body.userId,
        apartmentId: req.body.apartmentId,
        rentRequestId: req.body.rentRequestId,
        formId: req.body.formId,
        userStatus: req.body.userStatus,
        answers: req.body.answers.map((answer) => new feedbackModels.singleAnswerModel(answer)),
    });
 
    newAnswers.save(async function (err, answers) {
        if (err) return res.status(400).send({ error: 'answers failed to be saved' });
        try {
            await feedbackModels.pendingFeedbackModel.updateOne(
                {
                    rentRequest: new ObjectId(req.body.rentRequestId),
                    userStatus: req.body.userStatus,
                    userId: req.body.userId,
                },
                {
                    $set: {
                        status: PENDING_FEEDBACK_STATUS.ANSWERED,
                    },
                },
            );
        } catch {
            // la reponse a bien été ajouté mais le pendingfeeback a pas été updated
        }
        return res.status(200).send({
            message: 'answers successfully added',
            data: answers,
        });
    });
};
 
module.exports = { getActiveFeedbackForm, getPendingFeebacks, findPendingFeedbackById, submitFormAnswer };
